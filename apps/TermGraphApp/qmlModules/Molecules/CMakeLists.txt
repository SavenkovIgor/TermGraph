    PROPERTIES
        AUTOMOC ON
        CXX_STANDARD 20
        CXX_STANDARD_REQUIRED ON)

    URI Molecules
    VERSION 1.0
    IMPORTS StyleInfo Tools Atoms
    QML_FILES

        BurgerButton.qml
        DefaultHeader.qml
        EdgeHardeningDrawer.qml
        EmptyView.qml
        HelpCard.qml
        HelpView.qml
        LabelPair.qml
        LoadingIcon.qml
        LoadingInProcess.qml
        MainAppDrawer.qml
        NodesScene.qml
        NotificationDrawer.qml
        Page.qml
        SearchDrawer.qml
        SideMenuHeader.qml
        StdDialog.qml
        TagEditorArea.qml
        TermInfoDrawer.qml
        TextArea.qml
        TextField.qml)
